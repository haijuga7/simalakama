local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local Lighting = game:GetService("Lighting")
local StarterGui = game:GetService("StarterGui")
local RunService = game:GetService("RunService")
local Terrain = workspace:FindFirstChildOfClass("Terrain")

if not _G.Settings then
    _G.Settings = {
        Graphics = {
            FPS = 120,
            QualityLevel = 1,
            MeshPartDetail = 1,
            WaterQuality = 0,
            ShadowQuality = 0,
            MaterialQuality = 0,
            RemoveSky = true,
            RemoveAtmosphere = true,
            RemoveClouds = true
        },
        Optimizations = {
            RemoveClothing = true,
            RemoveAccessories = true,
            RemoveParticles = true,
            RemoveEffects = true,
            RemoveTextures = true,
            RemoveMeshes = false,
            RemoveExplosions = true,
            RemovePostEffects = true,
            RemoveLights = true,
            RemoveDecals = true
        },
        Notifications = true
    }
end

-- Notification system using Roblox's built-in notifications
local function showNotification(title, text, duration)
    if not _G.Settings.Notifications then return end
    
    StarterGui:SetCore("SendNotification", {
        Title = title,
        Text = text,
        Duration = duration or 5
    })
end

-- Core optimization functions
local function optimizeGraphics()
    showNotification("FPS Booster", "Starting optimizations...", 2)
    task.wait(0.5)
    
    -- Set FPS
    local success, err = pcall(function()
        if setfpscap then
            showNotification("FPS Booster", "Setting FPS to " .. _G.Settings.Graphics.FPS, 1)
            setfpscap(_G.Settings.Graphics.FPS)
        else
            showNotification("FPS Booster", "FPS Cap (setfpscap) not supported by executor.", 1)
        end
    end)
    if not success then
        showNotification("FPS Booster", "Error setting FPS: " .. err, 1)
    end
    task.wait(0.5)
    
    -- Optimize rendering
    showNotification("FPS Booster", "Optimizing rendering quality...", 1)
    settings().Rendering.QualityLevel = _G.Settings.Graphics.QualityLevel
    settings().Rendering.MeshPartDetailLevel = Enum.MeshPartDetailLevel.Level01
    task.wait(0.5)
    
    -- Optimize lighting
    showNotification("FPS Booster", "Removing shadows and fog...", 1)
    Lighting.GlobalShadows = false
    Lighting.ShadowSoftness = 0
    Lighting.FogEnd = 9e9
    task.wait(0.5)
    
    -- Optimize terrain
    if Terrain then
        showNotification("FPS Booster", "Optimizing water effects...", 1)
        Terrain.WaterWaveSize = _G.Settings.Graphics.WaterQuality
        Terrain.WaterWaveSpeed = 0
        Terrain.WaterReflectance = 0
        Terrain.WaterTransparency = 0
        if pcall(sethiddenproperty, Terrain, "Decoration", false) then
            -- Decoration hidden successfully
        else
            showNotification("FPS Booster", "Terrain decoration removal not supported.", 1)
        end
        task.wait(0.5)
    end

    -- Aggressively remove Sky, Atmosphere, and Clouds
    if _G.Settings.Graphics.RemoveSky then
        showNotification("FPS Booster", "Removing sky elements...", 1)
        for _, v in pairs(Lighting:GetDescendants()) do
            if v:IsA("Sky") or v:IsA("Atmosphere") or v:IsA("Clouds") then
                v:Destroy()
            end
        end
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("Sky") or v:IsA("Atmosphere") or v:IsA("Clouds") then
                v:Destroy()
            end
        end
        task.wait(0.5)
    end
end

-- Fungsi bantu untuk cek apakah instance berada di dalam objek dengan nama tertentu (obor, senter, dll)
local function isInAllowedLightSource(instance)
    local current = instance
    while current and current.Parent do
        local name = current.Name:lower()
        if name:find("torch") or name:find("obor") or name:find("flashlight") or name:find("senter") or name:find("light") then
            return true
        end
        current = current.Parent
    end
    return false
end

local function optimizeInstance(instance)
    if not instance then return end
    
    -- Remove clothing and accessories
    if _G.Settings.Optimizations.RemoveClothing and (instance:IsA("Clothing") or instance:IsA("SurfaceAppearance")) then
        instance:Destroy()
        return
    end
    
    -- Remove particles and effects
    if _G.Settings.Optimizations.RemoveParticles and (instance:IsA("ParticleEmitter") or instance:IsA("Trail") or instance:IsA("Smoke") or instance:IsA("Fire") or instance:IsA("Sparkles")) then
        instance.Enabled = false
        return
    end
    
    -- Remove textures
    if _G.Settings.Optimizations.RemoveTextures and instance:IsA("Texture") then
        instance.Transparency = 1
        return
    end
    
    -- Remove meshes
    if _G.Settings.Optimizations.RemoveMeshes and instance:IsA("SpecialMesh") then
        instance.MeshId = ""
        return
    end
    
    -- Remove explosions
    if _G.Settings.Optimizations.RemoveExplosions and instance:IsA("Explosion") then
        instance.BlastPressure = 1
        instance.BlastRadius = 1
        instance.Visible = false
        return
    end
    
    -- Remove post effects
    if _G.Settings.Optimizations.RemovePostEffects and instance:IsA("PostEffect") then
        instance.Enabled = false
        return
    end

    -- Remove Lights (tetapi kecuali yang berasal dari obor/senter)
    if _G.Settings.Optimizations.RemoveLights and (instance:IsA("SpotLight") or instance:IsA("PointLight") or instance:IsA("SurfaceLight")) then
        if not isInAllowedLightSource(instance) then
            instance.Enabled = false
            -- Jika ingin menghapus lampu yang tidak penting, uncomment baris berikut:
            -- instance:Destroy()
        end
        return
    end

    -- Remove Beams
    if _G.Settings.Optimizations.RemoveEffects and instance:IsA("Beam") then
        instance.Enabled = false
        return
    end

    -- Remove Decals
    if _G.Settings.Optimizations.RemoveDecals and instance:IsA("Decal") then
        instance.Transparency = 1
        instance:Destroy()
        return
    end
    
    -- Optimize parts
    if instance:IsA("BasePart") then
        instance.Material = Enum.Material.Plastic
        instance.Reflectance = 0
    end

    -- Remove Sky, Atmosphere, Clouds jika diaktifkan
    if _G.Settings.Graphics.RemoveSky and (instance:IsA("Sky") or instance:IsA("Atmosphere") or instance:IsA("Clouds")) then
        instance:Destroy()
        return
    end
end

-- Initialize optimizations
local function initialize()
    -- Show initial loader
    showNotification("FPS Booster", "Loading FPS Booster...", 1)
    task.wait(3)

    local success, err = pcall(function()
        optimizeGraphics()
        
        -- Optimize existing instances
        showNotification("FPS Booster", "Optimizing existing game objects...", 1)
        for _, instance in pairs(game:GetDescendants()) do
            optimizeInstance(instance)
        end
        task.wait(0.5)
        
        -- Optimize new instances
        showNotification("FPS Booster", "Monitoring new game objects...", 1)
        game.DescendantAdded:Connect(optimizeInstance)
        task.wait(0.5)
        
        -- Final progress notification
        showNotification("FPS Booster", "Optimizations completed successfully!", 1)
        task.wait(2)

        -- Show final message
        showNotification("FPS Booster", "FPS BOOSTED BY BLOXSTORE", 1)

    end)
    if not success then
        showNotification("FPS Booster", "Script Failed: " .. err, 1)
    end
end

-- Start optimizations
initialize()
