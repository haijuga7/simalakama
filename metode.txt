local metode = {}

function metode.method1(character, hrp, targetPos)
    if not hrp then return false end
    hrp.CFrame = CFrame.new(targetPos)
    task.wait(0.1)
    return (hrp.Position - targetPos).Magnitude <= 5
end

function metode.method2(character, hrp, targetPos)
    if not hrp then return false end
    local bp = Instance.new("BodyPosition")
    bp.Parent = hrp
    bp.MaxForce = Vector3.new(10000, 10000, 10000)
    bp.Position = targetPos
    task.wait(0.3)
    local success = (hrp.Position - targetPos).Magnitude <= 5
    bp:Destroy()
    return success
end

function metode.method3(character, hrp, targetPos)
    if not hrp then return false end
    hrp.CFrame = CFrame.new(targetPos)
    hrp.Anchored = true
    task.wait(0.1)
    local success = (hrp.Position - targetPos).Magnitude <= 5
    hrp.Anchored = false
    return success
end

function metode.method4(character, hrp, targetPos)
    if not hrp then return false end
    local tweenInfo = TweenInfo.new(0.5, Enum.EasingStyle.Linear)
    local tween = game:GetService("TweenService"):Create(hrp, tweenInfo, {CFrame = CFrame.new(targetPos)})
    tween:Play()
    task.wait(0.6)
    return (hrp.Position - targetPos).Magnitude <= 5
end

function metode.method5(character, hrp, targetPos)
    if not character or not character.PrimaryPart then return false end
    character:SetPrimaryPartCFrame(CFrame.new(targetPos))
    task.wait(0.1)
    return (hrp.Position - targetPos).Magnitude <= 5
end

return metode
